# 3. Notation

>>> seq('c', 'o', 'o', 'l')
<'c', 'o', 'o', 'l'>

>>> seq(epsilon, emptyset)

>>> len(seq('c', 'o', 'o', 'l'))
4

# 5. Four functions
>>> letters('cool')
<'c', 'o', 'o', 'l'>

{'c', 'l', 'o'}
>>> words('an orange cat')
<'an', 'orange', 'cat'>
>>> vocab('an orange cat')
{'an', 'cat', 'orange'}

# 6. Variables
>>> x = letters('dog')
>>> x
<'d', 'o', 'g'>

...     y

name 'y' is not defined

#--  String and set operations

# 7. Concatenation
>>> seq('c', 'a', 't') * seq('f', 'i', 's', 'h')
<'c', 'a', 't', 'f', 'i', 's', 'h'>
>>> letters('cat') * epsilon
<'c', 'a', 't'>

# 8. Set operations
>>> alphabet('abc') + alphabet('bcd')
{'a', 'b', 'c', 'd'}
>>> alphabet('abc') & alphabet('bcd')
{'b', 'c'}
>>> alphabet('abc') - alphabet('bcd')
{'a'}
>>> 'a' in alphabet('abc')
True
>>> alphabet('eo') <= alphabet('aeiou')
True
