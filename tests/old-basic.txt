
###  Section 7.3  A linguistic calculator

>>> 2 + 2
4

>>> 2 * 4
8

>>> 10 ** 2
100

>>> x = 42

>>> x
42

>>> x/2 + 1
22.0

>>> A = {'a', 'b', 'c'}
>>> B = {'c', 'd'}

>>> A | B == {'a', 'b', 'c', 'd'}
True
>>> A & B == {'c'}
True
>>> A - B == {'a', 'b'}
True

>>> 'a' in A
True
>>> {'a', 'b'} <= A
True

>>> x == 42
True

>>> 2 + 2 == 4
True

>>> {'a', 'b', 'c'} == {'c', 'a', 'b'}
True

>>> N = {0, 1, 2, 3, 4}
>>> {x for x in N if x < 3} == {0, 1, 2}
True

>>> ('a', 'b', 'c') == ('c', 'a', 'b')
False

>>> ('a', 'b') + ('c', 'd')
('a', 'b', 'c', 'd')

>>> def S(x):
...     return x + 1
...

>>> S(4)
5

>>> S(x)
43
